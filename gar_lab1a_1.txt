create table employees (
    full_name  varchar(50) not null,
    position varchar(30) not null,
    height tinyint unsigned null,
    weight FLOAT null
);
create table patients (
    full_name varchar(50) not null,
    diagnosis varchar(40) not null,
    height tinyint unsigned not null,
    weight FLOAT not null
);
create table accident (
    date DATETIME not null,
    full_name_injured varchar(50) not null,
    full_name_ initiator varchar(50) not null,
    degree_of_damage varchar(3) not null,
    start_data_healing DATETIME not null default current_timestamp,
    healing_period DATETIME not null
);
